# 边缘设备模型参数表现数据采集方案

## 目标

对不同参数组合（量化精度、生成参数、LoRA配置等）在边缘设备上运行时，系统性地采集**推理速度、内存占用、输出质量、功耗/CPU占用**四类指标，用于筛选最优参数配置。

---

## 一、采集指标体系

| 类别 | 指标 | 单位 | 说明 |
|------|------|------|------|
| **推理速度** | 首 token 延迟（TTFT） | ms | 从输入到第一个 token 输出的时间 |
| | 生成吞吐量 | tokens/s | 每秒生成的 token 数 |
| | 总推理时间 | ms | 完整生成一条输出的耗时 |
| **内存占用** | 模型加载内存 | MB | 模型加载完成后的内存基线 |
| | 推理峰值内存 | MB | 单次推理过程中的最大内存 |
| | GPU 显存（若有） | MB | CUDA 显存占用 |
| **输出质量** | BLEU / ROUGE 分数 | 0~1 | 与标准答案的文本相似度 |
| | 准确率 | % | 分类/问答任务的正确率 |
| | 人工评分 | 1~5 | 主观质量评估（可选） |
| **资源占用** | CPU 使用率 | % | 推理期间平均 CPU 占用 |
| | 功耗 | W | 设备推理时的瓦特数（需硬件支持）|
| | 推理温度 | °C | 设备核心温度（防止热降频） |

## 二、采集代码

### 2.1 核心采集模块

```python
import time
import psutil
import threading
import torch
from dataclasses import dataclass, field
from typing import Optional

@dataclass
class BenchmarkResult:
    # 参数配置
    model_name: str
    quantization: str          # "fp16" / "int8" / "int4"
    max_new_tokens: int
    do_sample: bool
    temperature: float
    top_p: float
    num_beams: int

    # 速度指标
    ttft_ms: float = 0.0       # 首 token 延迟
    total_time_ms: float = 0.0
    tokens_per_second: float = 0.0
    output_tokens: int = 0

    # 内存指标
    ram_baseline_mb: float = 0.0
    ram_peak_mb: float = 0.0
    gpu_mem_mb: float = 0.0

    # 资源指标
    cpu_avg_percent: float = 0.0

    # 质量指标
    output_text: str = ""
    bleu_score: float = 0.0
    rouge_score: float = 0.0


class DeviceMonitor:
    """后台线程持续采集 CPU、内存"""
    def __init__(self):
        self.cpu_samples = []
        self.ram_samples = []
        self._running = False
    def start(self):
        self._running = True
        self._thread = threading.Thread(target=self._collect, daemon=True)
        self._thread.start()
    def stop(self):
        self._running = False
        self._thread.join()
    def _collect(self):
        while self._running:
            self.cpu_samples.append(psutil.cpu_percent(interval=0.1))
            self.ram_samples.append(psutil.virtual_memory().used / 1024 / 1024)
    @property
    def cpu_avg(self):
        return sum(self.cpu_samples) / len(self.cpu_samples) if self.cpu_samples else 0
    @property
    def ram_peak(self):
        return max(self.ram_samples) if self.ram_samples else 0


def benchmark_single(model, tokenizer, prompt, config: dict) -> BenchmarkResult:
    """对单个参数配置跑一次基准测试"""

    result = BenchmarkResult(**config)

    # 记录内存基线
    result.ram_baseline_mb = psutil.virtual_memory().used / 1024 / 1024
    if torch.cuda.is_available():
        torch.cuda.reset_peak_memory_stats()

    inputs = tokenizer(prompt, return_tensors="pt").to(model.device)

    # 启动资源监控
    monitor = DeviceMonitor()
    monitor.start()

    # 记录首 token 延迟
    first_token_time = None
    start_time = time.perf_counter()

    def first_token_hook(module, input, output):
        nonlocal first_token_time
        if first_token_time is None:
            first_token_time = time.perf_counter()

    # 注册 hook 捕获首 token
    hook = model.register_forward_hook(first_token_hook)

    with torch.no_grad():
        output_ids = model.generate(
            **inputs,
            max_new_tokens=config["max_new_tokens"],
            do_sample=config["do_sample"],
            temperature=config.get("temperature", 1.0),
            top_p=config.get("top_p", 1.0),
            num_beams=config.get("num_beams", 1),
            use_cache=True,
        )

    end_time = time.perf_counter()
    hook.remove()
    monitor.stop()

    # 计算速度指标
    result.total_time_ms = (end_time - start_time) * 1000
    result.ttft_ms = (first_token_time - start_time) * 1000 if first_token_time else 0
    result.output_tokens = output_ids.shape[1] - inputs["input_ids"].shape[1]
    result.tokens_per_second = result.output_tokens / (end_time - start_time)

    # 内存指标
    result.ram_peak_mb = monitor.ram_peak
    if torch.cuda.is_available():
        result.gpu_mem_mb = torch.cuda.max_memory_allocated() / 1024 / 1024

    # 资源指标
    result.cpu_avg_percent = monitor.cpu_avg

    # 解码输出
    generated_ids = output_ids[0][inputs["input_ids"].shape[1]:]
    result.output_text = tokenizer.decode(generated_ids, skip_special_tokens=True)

    return result
```

### 2.2 输出质量评估

```python
from nltk.translate.bleu_score import sentence_bleu, SmoothingFunction
from rouge_score import rouge_scorer

def evaluate_quality(prediction: str, reference: str) -> dict:
    """计算 BLEU 和 ROUGE 分数"""

    # BLEU
    ref_tokens = reference.split()
    pred_tokens = prediction.split()
    bleu = sentence_bleu(
        [ref_tokens], pred_tokens,
        smoothing_function=SmoothingFunction().method1
    )

    # ROUGE
    scorer = rouge_scorer.RougeScorer(["rouge1", "rouge2", "rougeL"], use_stemmer=False)
    rouge = scorer.score(reference, prediction)

    return {
        "bleu": round(bleu, 4),
        "rouge1": round(rouge["rouge1"].fmeasure, 4),
        "rouge2": round(rouge["rouge2"].fmeasure, 4),
        "rougeL": round(rouge["rougeL"].fmeasure, 4),
    }
```

### 2.3 批量参数组合扫描

```python
import itertools
import pandas as pd
from transformers import AutoModelForCausalLM, AutoTokenizer, BitsAndBytesConfig

# 定义要扫描的参数网格
PARAM_GRID = {
    "quantization":    ["fp16", "int8", "int4"],
    "max_new_tokens":  [64, 128, 256],
    "do_sample":       [False, True],
    "temperature":     [0.7],          # do_sample=True 时才生效
    "top_p":           [0.8],
    "num_beams":       [1],
}

# 测试用的 prompt 集合（覆盖不同任务类型）
TEST_PROMPTS = [
    {"input": "请用一句话介绍人工智能。",         "reference": "人工智能是模拟人类智能的技术。"},
    {"input": "1+1等于多少？",                   "reference": "2"},
    {"input": "用Python写一个冒泡排序函数。",     "reference": ""},  # 代码类，只看速度
]

def load_model_with_quant(model_name, quant_type):
    """按量化类型加载模型"""
    if quant_type == "int4":
        bnb_config = BitsAndBytesConfig(
            load_in_4bit=True,
            bnb_4bit_compute_dtype=torch.float16,
            bnb_4bit_use_double_quant=True,
            bnb_4bit_quant_type="nf4",
        )
        return AutoModelForCausalLM.from_pretrained(
            model_name, quantization_config=bnb_config, device_map="auto"
        )
    elif quant_type == "int8":
        return AutoModelForCausalLM.from_pretrained(
            model_name, load_in_8bit=True, device_map="auto"
        )
    else:  # fp16
        return AutoModelForCausalLM.from_pretrained(
            model_name, torch_dtype=torch.float16, device_map="auto"
        )


def run_benchmark(model_name="Qwen/Qwen2.5-1.5B-Instruct"):
    tokenizer = AutoTokenizer.from_pretrained(model_name)
    results = []

    # 遍历所有参数组合
    keys = list(PARAM_GRID.keys())
    for values in itertools.product(*PARAM_GRID.values()):
        config = dict(zip(keys, values))

        # do_sample=False 时跳过 temperature 的多余组合
        if not config["do_sample"] and config["temperature"] != 0.7:
            continue

        print(f"测试配置: {config}")
        model = load_model_with_quant(model_name, config["quantization"])

        for prompt_item in TEST_PROMPTS:
            config["model_name"] = model_name
            result = benchmark_single(model, tokenizer, prompt_item["input"], config)

            # 计算质量分数
            if prompt_item["reference"]:
                quality = evaluate_quality(result.output_text, prompt_item["reference"])
                result.bleu_score = quality["bleu"]
                result.rouge_score = quality["rougeL"]

            results.append(result.__dict__)

        del model
        torch.cuda.empty_cache() if torch.cuda.is_available() else None

    # 保存结果
    df = pd.DataFrame(results)
    df.to_csv("benchmark_results.csv", index=False)
    print("结果已保存到 benchmark_results.csv")
    return df
```

## 三、结果分析与可视化

```python
import matplotlib.pyplot as plt
import matplotlib
matplotlib.rcParams['font.family'] = 'SimHei'  # 支持中文

def analyze_results(csv_path="benchmark_results.csv"):
    df = pd.read_csv(csv_path)

    fig, axes = plt.subplots(2, 2, figsize=(14, 10))
    fig.suptitle("边缘设备模型参数表现对比", fontsize=16)

    # 1. 不同量化精度的推理速度
    speed_by_quant = df.groupby("quantization")["tokens_per_second"].mean()
    axes[0, 0].bar(speed_by_quant.index, speed_by_quant.values, color=["#2196F3","#4CAF50","#FF9800"])
    axes[0, 0].set_title("量化精度 vs 推理速度")
    axes[0, 0].set_ylabel("tokens/s")

    # 2. 内存占用对比
    mem_by_quant = df.groupby("quantization")["ram_peak_mb"].mean()
    axes[0, 1].bar(mem_by_quant.index, mem_by_quant.values, color=["#2196F3","#4CAF50","#FF9800"])
    axes[0, 1].set_title("量化精度 vs 峰值内存")
    axes[0, 1].set_ylabel("MB")

    # 3. max_new_tokens 对总时间的影响
    time_by_tokens = df.groupby("max_new_tokens")["total_time_ms"].mean()
    axes[1, 0].plot(time_by_tokens.index, time_by_tokens.values, marker="o", color="#9C27B0")
    axes[1, 0].set_title("max_new_tokens vs 总耗时")
    axes[1, 0].set_ylabel("ms")

    # 4. 速度 vs 质量散点图
    axes[1, 1].scatter(df["tokens_per_second"], df["bleu_score"],
                       c=df["quantization"].map({"fp16":0,"int8":1,"int4":2}),
                       cmap="viridis", alpha=0.6, s=60)
    axes[1, 1].set_title("推理速度 vs 输出质量")
    axes[1, 1].set_xlabel("tokens/s")
    axes[1, 1].set_ylabel("BLEU score")

    plt.tight_layout()
    plt.savefig("benchmark_report.png", dpi=150)
    print("图表已保存到 benchmark_report.png")

    # 打印最优配置
    print("\n=== 最优配置（速度最快）===")
    print(df.loc[df["tokens_per_second"].idxmax()])

    print("\n=== 最优配置（质量最高）===")
    print(df.loc[df["bleu_score"].idxmax()])

    print("\n=== 最优配置（内存最小）===")
    print(df.loc[df["ram_peak_mb"].idxmin()])
```

## 五、采集流程总结

```
1. 定义参数网格（量化/生成参数组合）
        ↓
2. 逐个加载模型配置
        ↓
3. 后台线程持续采集 CPU/内存
        ↓
4. 运行推理，记录首token延迟、吞吐量
        ↓
5. 计算 BLEU/ROUGE 质量分数
        ↓
6. 汇总写入 CSV
        ↓
7. 可视化对比，输出最优配置建议
```
